            // Assign our favorite number (which is 42) to a variable named "favoriteNumber". (That is a good variable name.)
            var favoriteNumber = 42;
            // At this point, favoriteNumber is an alias to a location in the computer's memory
            // which contains the integer value 42.
            alert(favoriteNumber);


            // Assign our favorite teacher (which is Alan) to a variable named "ta". (That is a bad variable name.)
            var ta = "Alan";
            // At this point, ta is an alias to a location in the computer's memory
            // which contains the text (or string) value "Alan".
            alert(ta);


            // The next non-comment line find the text area in the DOM
            // and assign it some new text; all in one line of code.
            document.getElementById("taMain").value = "Hello, world";
            // That works, but we could make it more easily read and understood
            // by breaking it up into its component parts; which we will no do:

            // 1. Assign the message we want to display to a variable named "message". (This is a good variable name.)
            var message = "Hello, world.";

            // 2. Find the text area in the DOM and assign it to a variable named "targetTextArea".
            var targetTextArea = document.getElementById("taMain");

            // 3. Finally, assign the message to the text (the "value" attribute) of the DOM's text area.
            targetTextArea.value = message;
         }
